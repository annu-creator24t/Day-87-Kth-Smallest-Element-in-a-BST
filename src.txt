class Solution {
    public int kthSmallest(TreeNode root, int k) {
        Stack<TreeNode> stack = new Stack<>();
        
        while (true) {
            // Go to the leftmost node
            while (root != null) {
                stack.push(root);
                root = root.left;
            }
            
            root = stack.pop();
            k--; // Visit this node
            if (k == 0) {
                return root.val; // kth smallest found
            }
            
            root = root.right; // Go right
        }
    }
}
